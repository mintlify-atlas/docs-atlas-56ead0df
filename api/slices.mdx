## User Slice

Manages user authentication and profile state.

**Source**: `~/workspace/source/src/app/redux/slice/userSlice.ts`

### State Shape

```typescript
interface UserState {
  user: IUser | null;
  isAuthenticated: boolean;
  loading: boolean;
}
```

### Actions

#### setUser

Set the current user and authentication status.

```typescript
import { setUser } from '@/app/redux/slice/userSlice';

dispatch(setUser({
  id: '123',
  name: 'John Doe',
  email: 'john@example.com',
}));
```

#### setLoading

Set loading state for user operations.

```typescript
import { setLoading } from '@/app/redux/slice/userSlice';

dispatch(setLoading(true));
```

#### logout

Clear user state and mark as unauthenticated.

```typescript
import { logout } from '@/app/redux/slice/userSlice';

dispatch(logout());
```

### Usage Example

```typescript
import { useSelector, useDispatch } from 'react-redux';
import { RootState } from '@/app/redux/store';
import { setUser, logout } from '@/app/redux/slice/userSlice';

function AuthComponent() {
  const dispatch = useDispatch();
  const { user, isAuthenticated, loading } = useSelector(
    (state: RootState) => state.user
  );

  const handleLogin = (userData: IUser) => {
    dispatch(setUser(userData));
  };

  const handleLogout = () => {
    dispatch(logout());
  };

  return (
    <div>
      {isAuthenticated ? (
        <div>
          <p>Welcome, {user?.name}</p>
          <button onClick={handleLogout}>Logout</button>
        </div>
      ) : (
        <button onClick={() => handleLogin(userData)}>Login</button>
      )}
    </div>
  );
}
```

## Filter Slice

Manages search queries, filters, and recent searches.

**Source**: `~/workspace/source/src/app/redux/slice/filterSlice.ts`

### State Shape

```typescript
interface FilterState {
  searchQuery: string;
  recentSearches: string[];
  selectedFixerNames: string[];
  selectedCities: string[];
  selectedJobTypes: string[];
  isAutoSelectedJobType: boolean;
  isAutoSelectedCity: boolean;
  sidebarOpen: boolean;
}
```

### Actions

#### setSearchQuery

Set the current search query.

```typescript
import { setSearchQuery } from '@/app/redux/slice/filterSlice';

dispatch(setSearchQuery('plumber'));
```

#### addRecentSearch

Add a search to recent searches (max 5).

```typescript
import { addRecentSearch } from '@/app/redux/slice/filterSlice';

dispatch(addRecentSearch('electrician Brooklyn'));
```

#### toggleCity / toggleJobType / toggleFixerName

Toggle filter selections.

```typescript
import { toggleCity, toggleJobType } from '@/app/redux/slice/filterSlice';

dispatch(toggleCity('New York'));
dispatch(toggleJobType('plumbing'));
```

#### autoSelectJobType / autoSelectCity

Auto-select filters (marked as automatic).

```typescript
import { autoSelectJobType } from '@/app/redux/slice/filterSlice';

dispatch(autoSelectJobType('plumbing'));
```

#### resetFilters

Clear all filters and search query.

```typescript
import { resetFilters } from '@/app/redux/slice/filterSlice';

dispatch(resetFilters());
```

### Selectors

```typescript
import {
  selectSearchQuery,
  selectSelectedCities,
  selectSelectedJobTypes,
} from '@/app/redux/slice/filterSlice';

const searchQuery = useSelector(selectSearchQuery);
const cities = useSelector(selectSelectedCities);
```

### Usage Example

```typescript
import { useSelector, useDispatch } from 'react-redux';
import {
  setSearchQuery,
  toggleCity,
  resetFilters,
  selectSearchQuery,
  selectSelectedCities,
} from '@/app/redux/slice/filterSlice';

function SearchFilters() {
  const dispatch = useDispatch();
  const searchQuery = useSelector(selectSearchQuery);
  const selectedCities = useSelector(selectSelectedCities);

  return (
    <div>
      <input
        value={searchQuery}
        onChange={(e) => dispatch(setSearchQuery(e.target.value))}
      />
      <button onClick={() => dispatch(toggleCity('New York'))}>
        New York {selectedCities.includes('New York') ? '✓' : ''}
      </button>
      <button onClick={() => dispatch(resetFilters())}>
        Reset All
      </button>
    </div>
  );
}
```

## Best Practices

### Use Typed Selectors

```typescript
import { useSelector } from 'react-redux';
import { RootState } from '@/app/redux/store';

const user = useSelector((state: RootState) => state.user.user);
```

### Batch Related Actions

```typescript
import { useDispatch } from 'react-redux';
import { setSearchQuery, toggleCity } from '@/app/redux/slice/filterSlice';

const dispatch = useDispatch();

// Batch related actions
dispatch(setSearchQuery('plumber'));
dispatch(toggleCity('New York'));
```

### Use Memoized Selectors

```typescript
import { createSelector } from '@reduxjs/toolkit';
import { RootState } from '@/app/redux/store';

const selectActiveFiltersCount = createSelector(
  [(state: RootState) => state.filters],
  (filters) =>
    filters.selectedCities.length +
    filters.selectedJobTypes.length +
    filters.selectedFixerNames.length
);
```
