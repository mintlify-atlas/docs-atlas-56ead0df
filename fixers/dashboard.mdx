## Dashboard Overview

The Fixer Dashboard is your central hub for managing your business on Servineo. Access it at `/fixer/dashboard`.

## Dashboard Layout

The dashboard uses a two-column layout with sidebar navigation:

```tsx
import { useAppSelector } from '@/app/redux/hooks';

export default function FixerDashboardPage() {
  const { user: reduxUser } = useAppSelector((state) => state.user);
  const [activeTab, setActiveTab] = useState<Tab>('offers');

  return (
    <div className='container mx-auto max-w-6xl p-4 space-y-6'>
      {/* Profile Header */}
      <ProfileHeader user={user} />
      
      <div className='grid grid-cols-1 lg:grid-cols-4 gap-6'>
        {/* Sidebar Navigation */}
        <SidebarNav activeTab={activeTab} setActiveTab={setActiveTab} />
        
        {/* Main Content */}
        <MainContent activeTab={activeTab} />

  );
}
```

## Editable Description

Fixers can edit their profile description with validation:

```typescript
import { z } from 'zod';

const descriptionSchema = z.object({
  description: z
    .string()
    .max(255, 'La descripción no puede tener más de 255 caracteres')
    .min(10, 'La descripción debe tener al menos 10 caracteres')
    .optional()
    .or(z.literal('')),
});
```

```tsx
const [editingDescription, setEditingDescription] = useState(false);
const [descriptionValue, setDescriptionValue] = useState('');
const [updateDescription] = useUpdateDescriptionMutation();

const handleSaveDescription = async () => {
  const result = descriptionSchema.safeParse({ description: descriptionValue });
  
  if (!result.success) {
    setDescriptionError(result.error.issues[0].message);
    return;
  }
  
  await updateDescription({ id: user._id, description: descriptionValue });
  setEditingDescription(false);
};
```
